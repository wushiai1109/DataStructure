class Solution {
    public void moveZeroes(int[] nums) {

        int k = 0;
        for (int i = 0; i < nums.length; i++)
            if (nums[i] != 0)
                if (k != i)
                    swap(nums, k++, i);
                else
                    k++;
    }

    private void swap(int[] nums, int i, int j) {
        int t = nums[i];
        nums[i] = nums[j];
        nums[j] = t;
    }
}
